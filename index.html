        document.getElementById('q-counter').innerText = `${currentIdx + 1} / 15`;

        document.getElementById('q-title').innerText = q.t;

        

        const btnA = document.getElementById('opt-a');

        const btnB = document.getElementById('opt-b');

        

        btnA.innerText = "A. " + q.a;

        btnA.onclick = () => handleAnswer(q.aVal);

        

        btnB.innerText = "B. " + q.b;

        btnB.onclick = () => handleAnswer(q.bVal);


        // 更新进度条

        const percent = ((currentIdx) / 15) * 100;

        document.getElementById('progress-bar').style.width = percent + '%';

    }


    function handleAnswer(val) {

        scores[val]++;

        currentIdx++;

        

        if (currentIdx < questions.length) {

            renderQuestion();

        } else {

            document.getElementById('progress-bar').style.width = '100%';

            setTimeout(showResult, 300);

        }

    }


    function showResult() {

        document.getElementById('screen-question').classList.remove('active');

        document.getElementById('progress-container').style.display = 'none';

        

        // 计算四字人格

        const dim1 = scores.E >= scores.I ? 'E' : 'I';

        const dim2 = scores.C >= scores.B ? 'C' : 'B';

        const dim3 = scores.S >= scores.F ? 'S' : 'F';

        const dim4 = scores.D >= scores.L ? 'D' : 'L';

        const finalType = dim1 + dim2 + dim3 + dim4;


        // 粗略归类逻辑

        let categoryKey = "贪吃"; // 默认兜底

        if (dim1 === 'E' && dim4 === 'D') categoryKey = "悍匪";

        else if (dim2 === 'B' && dim4 === 'L') categoryKey = "白月光";

        else if (dim1 === 'I' && dim2 === 'C') categoryKey = "吗喽";

        else if (dim2 === 'B' && dim4 === 'D') categoryKey = "贪吃";

        else if (dim1 === 'I' && dim4 === 'L') categoryKey = "白月光";

        else if (dim2 === 'C' && dim3 === 'F') categoryKey = "吗喽";

        else categoryKey = "悍匪"; // 剩余暴躁老哥一律打成悍匪


        const res = resultsData[categoryKey];


        // 渲染文案

        document.getElementById('r-type').innerText = finalType;

        document.getElementById('r-name').innerText = res.name;

        document.getElementById('r-desc').innerText = res.desc;

        document.getElementById('r-product-desc').innerText = res.productDesc;

        document.getElementById('r-product-name').innerText = res.productName;

        

        // 渲染立绘和购物链接

        document.getElementById('r-img').src = res.imgSrc;

        document.getElementById('r-shop-link').href = res.shopLink;


        document.getElementById('screen-result').classList.add('active');

    }
